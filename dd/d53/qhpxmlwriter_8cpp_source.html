113 void QhpXmlWriter::openPureHelper(const QCString &elementName,
114  const char * const attributes[], bool close)
115 {
116  m_backend << "<" << elementName;
117  if (attributes)
118  {
119  for (const char * const * walker = attributes;
120  walker[0]; walker += 2)
121  {
122  const char *const key = walker[0];
123  const char *const value = walker[1];
124  if (!value)
125  {
126  continue;
127  }
128  m_backend << " " << key << "=\"" << convertToXML(value) << "\"";
129  }
130  }
131 
132  if (close)
133  {
134  m_backend << " /";
135  }
136  m_backend << ">";
137 }
