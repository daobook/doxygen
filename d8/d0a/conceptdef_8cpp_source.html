283 void ConceptDefImpl::writeBriefDescription(OutputList &ol) const
284 {
285  if (hasBriefDescription())
286  {
287  std::unique_ptr<IDocParser> parser { createDocParser() };
288  std::unique_ptr<DocRoot> rootNode { validatingParseDoc(

292  if (rootNode && !rootNode->isEmpty())
293  {
294  ol.startParagraph();
295  ol.pushGeneratorState();
296  ol.disableAllBut(OutputGenerator::Man);
297  ol.writeString(" - ");

300  ol.pushGeneratorState();
301  ol.disable(OutputGenerator::RTF);
302  ol.writeString(" \n");
303  ol.enable(OutputGenerator::RTF);
304 
305  if (hasDetailedDescription())
306  {
307  ol.disableAllBut(OutputGenerator::Html);
308  ol.startTextLink(getOutputFileBase(),"details");
309  ol.parseText(theTranslator->trMore());
310  ol.endTextLink();
311  }
312  ol.popGeneratorState();
313  ol.endParagraph();
314  }
315  }
316  ol.writeSynopsis();

320 {

361 static QCString templateSpec(const ArgumentList &al)
362 {
363  TextStream t;
364  t << "template<";
365  bool first=true;
366  for (const auto &a : al)
367  {
368  if (!first) t << ", ";
369  t << a.type;
370  if (!a.name.isEmpty())
371  {
372  t << " " << a.name;
373  }
374  if (a.defval.length()!=0)
375  {
376  t << " " << a.defval;
377  }
378  first=false;
379  }
380  t << ">";
381  return t.str();

412 void ConceptDefImpl::writeDetailedDescription(OutputList &ol,const QCString &title) const
413 {
414  static bool repeatBrief = Config_getBool(REPEAT_BRIEF);
415  if (hasDetailedDescription())
416  {
417  ol.pushGeneratorState();
418  ol.disable(OutputGenerator::Html);
419  ol.writeRuler();
420  ol.popGeneratorState();
421 
422  ol.pushGeneratorState();
423  ol.disableAllBut(OutputGenerator::Html);
424  ol.writeAnchor(QCString(),"details");
425  ol.popGeneratorState();
426 
427  ol.startGroupHeader();
428  ol.parseText(title);
429  ol.endGroupHeader();
430 
431  ol.startTextBlock();
432  // repeat brief description
433  if (!briefDescription().isEmpty() && repeatBrief)
434  {
435  ol.generateDoc(briefFile(),briefLine(),this,0,briefDescription(),FALSE,FALSE,
436  QCString(),FALSE,FALSE,Config_getBool(MARKDOWN_SUPPORT));
437  }
438  if (!briefDescription().isEmpty() && repeatBrief &&
439  !documentation().isEmpty())
440  {
441  ol.pushGeneratorState();
442  ol.disable(OutputGenerator::Html);
443  ol.writeString("\n\n");
444  ol.popGeneratorState();
445  }
446 
447  // write documentation
448  if (!documentation().isEmpty())
449  {
450  ol.generateDoc(docFile(),docLine(),this,0,documentation(),TRUE,FALSE,
451  QCString(),FALSE,FALSE,Config_getBool(MARKDOWN_SUPPORT));
452  }
453 
454  writeSourceDef(ol,name());
455  ol.endTextBlock();
456  }

471 void ConceptDefImpl::writeDocumentation(OutputList &ol)
472 {
473  static bool generateTreeView = Config_getBool(GENERATE_TREEVIEW);
474  QCString pageTitle = theTranslator->trConceptReference(displayName());
475  startFile(ol,getOutputFileBase(),name(),pageTitle,HLI_ConceptVisible,!generateTreeView);
476 
477  // ---- navigation part
478  if (!generateTreeView)
479  {
480  if (getOuterScope()!=Doxygen::globalScope)
481  {
482  writeNavigationPath(ol);
483  }
484  ol.endQuickIndices();
485  }
486 
487  // ---- title part
488  startTitle(ol,getOutputFileBase(),this);
489  ol.parseText(pageTitle);
490  addGroupListToTitle(ol,this);
491  endTitle(ol,getOutputFileBase(),displayName());
492 
493  // ---- contents part
494 
495  ol.startContents();
496 
497  //---------------------------------------- start flexible part -------------------------------
498 
499  for (const auto &lde : LayoutDocManager::instance().docEntries(LayoutDocManager::Concept))
500  {
501  switch (lde->kind())
502  {
503  case LayoutDocEntry::BriefDesc:
504  writeBriefDescription(ol);
505  break;
506  case LayoutDocEntry::ConceptDefinition:
507  {
508  const LayoutDocEntrySection *ls = (const LayoutDocEntrySection*)lde.get();
509  writeDefinition(ol,ls->title(getLanguage()));
510  }
511  break;
512  case LayoutDocEntry::DetailedDesc:
513  {
514  const LayoutDocEntrySection *ls = (const LayoutDocEntrySection*)lde.get();
515  writeDetailedDescription(ol,ls->title(getLanguage()));
516  }
517  break;
518  case LayoutDocEntry::AuthorSection:
519  writeAuthorSection(ol);
520  break;
521  case LayoutDocEntry::ClassIncludes:
522  writeIncludeFiles(ol);
523  break;
524  case LayoutDocEntry::MemberDeclStart:
525  case LayoutDocEntry::NamespaceClasses:
526  case LayoutDocEntry::NamespaceInterfaces:
527  case LayoutDocEntry::NamespaceStructs:
528  case LayoutDocEntry::NamespaceExceptions:
529  case LayoutDocEntry::NamespaceConcepts:
530  case LayoutDocEntry::NamespaceNestedNamespaces:
531  case LayoutDocEntry::NamespaceNestedConstantGroups:
532  case LayoutDocEntry::MemberGroups:
533  case LayoutDocEntry::MemberDecl:
534  case LayoutDocEntry::MemberDeclEnd:
535  case LayoutDocEntry::MemberDefStart:
536  case LayoutDocEntry::NamespaceInlineClasses:
537  case LayoutDocEntry::MemberDef:
538  case LayoutDocEntry::MemberDefEnd:
539  case LayoutDocEntry::ClassInheritanceGraph:
540  case LayoutDocEntry::ClassNestedClasses:
541  case LayoutDocEntry::ClassCollaborationGraph:
542  case LayoutDocEntry::ClassAllMembersLink:
543  case LayoutDocEntry::ClassUsedFiles:
544  case LayoutDocEntry::ClassInlineClasses:
545  case LayoutDocEntry::FileClasses:
546  case LayoutDocEntry::FileConcepts:
547  case LayoutDocEntry::FileInterfaces:
548  case LayoutDocEntry::FileStructs:
549  case LayoutDocEntry::FileExceptions:
550  case LayoutDocEntry::FileNamespaces:
551  case LayoutDocEntry::FileConstantGroups:
552  case LayoutDocEntry::FileIncludes:
553  case LayoutDocEntry::FileIncludeGraph:
554  case LayoutDocEntry::FileIncludedByGraph:
555  case LayoutDocEntry::FileInlineClasses:
556  case LayoutDocEntry::FileSourceLink:
557  case LayoutDocEntry::GroupClasses:
558  case LayoutDocEntry::GroupConcepts:
559  case LayoutDocEntry::GroupInlineClasses:
560  case LayoutDocEntry::GroupNamespaces:
561  case LayoutDocEntry::GroupDirs:
562  case LayoutDocEntry::GroupNestedGroups:
563  case LayoutDocEntry::GroupFiles:
564  case LayoutDocEntry::GroupGraph:
565  case LayoutDocEntry::GroupPageDocs:
566  case LayoutDocEntry::DirSubDirs:
567  case LayoutDocEntry::DirFiles:
568  case LayoutDocEntry::DirGraph:
569  err("Internal inconsistency: member %d should not be part of "

572  }
573  }
574

576 
577  ol.endContents();
578 
579  endFileWithNavPath(this,ol);

582 void ConceptDefImpl::writeDeclarationLink(OutputList &ol,bool &found,const QCString &header,bool localNames) const
583 {
584  if (isLinkable())
585  {
586  if (!found) // first concept
587  {
588  ol.startMemberHeader("concepts");
589  if (!header.isEmpty())
590  {
591  ol.parseText(header);
592  }
593  else
594  {
595  theTranslator->trConcept(true,false);
596  }
597  ol.endMemberHeader();
598  ol.startMemberList();
599  found=TRUE;
600  }
601  ol.startMemberDeclaration();
602  ol.startMemberItem(anchor(),FALSE);
603  ol.writeString("concept ");
604  QCString cname = displayName(!localNames);
605  ol.insertMemberAlign();
606  if (isLinkable())
607  {
608  ol.writeObjectLink(getReference(),
609  getOutputFileBase(),
610  anchor(),
611  cname
612  );
613  }
614  else
615  {
616  ol.startBold();
617  ol.docify(cname);
618  ol.endBold();
619  }
620  ol.endMemberItem();
621  // add the brief description if available
622  if (!briefDescription().isEmpty() && Config_getBool(BRIEF_MEMBER_DESC))
623  {
624  std::unique_ptr<IDocParser> parser { createDocParser() };
625  std::unique_ptr<DocRoot> rootNode { validatingParseDoc(
626  *parser.get(),briefFile(),briefLine(),this,0,
627  briefDescription(),FALSE,FALSE,
628  QCString(),TRUE,FALSE,Config_getBool(MARKDOWN_SUPPORT)) };
629  if (rootNode && !rootNode->isEmpty())
630  {
631  ol.startMemberDescription(anchor());
632  ol.writeDoc(rootNode.get(),this,0);
633  ol.endMemberDescription();
634  }
635  }
636  ol.endMemberDeclaration(anchor(),QCString());
637  }
638 }
