128 void ManDocVisitor::visit(DocStyleChange *s)
129 {
130  if (m_hide) return;
131  switch (s->style())
132  {
133  case DocStyleChange::Bold:
134  if (s->enable()) m_t << "\\fB"; else m_t << "\\fP";
135  m_firstCol=FALSE;
136  break;
137  case DocStyleChange::S:
138  case DocStyleChange::Strike:
139  case DocStyleChange::Del:
140  /* not supported */
141  break;
142  case DocStyleChange::Underline: //underline is shown as emphasis
143  case DocStyleChange::Ins:
144  if (s->enable()) m_t << "\\fI"; else m_t << "\\fP";
145  m_firstCol=FALSE;
146  break;
147  case DocStyleChange::Italic:
148  if (s->enable()) m_t << "\\fI"; else m_t << "\\fP";
149  m_firstCol=FALSE;
150  break;
151  case DocStyleChange::Code:
152  if (s->enable()) m_t << "\\fC"; else m_t << "\\fP";
153  m_firstCol=FALSE;
154  break;
155  case DocStyleChange::Subscript:
156  if (s->enable()) m_t << "\\*<"; else m_t << "\\*> ";
157  m_firstCol=FALSE;
158  break;
159  case DocStyleChange::Superscript:
160  if (s->enable()) m_t << "\\*{"; else m_t << "\\*} ";
161  m_firstCol=FALSE;
162  break;
163  case DocStyleChange::Center:
164  /* not supported */
165  break;
166  case DocStyleChange::Small:
167  /* not supported */
168  break;
169  case DocStyleChange::Preformatted:
170  if (s->enable())
171  {
172  if (!m_firstCol) m_t << "\n";
173  m_t << ".PP\n";
174  m_t << ".nf\n";
175  m_insidePre=TRUE;
176  }
177  else
178  {
179  m_insidePre=FALSE;
180  if (!m_firstCol) m_t << "\n";
181  m_t << ".fi\n";
182  m_t << ".PP\n";
183  m_firstCol=TRUE;
184  }
185  break;
186  case DocStyleChange::Div: /* HTML only */ break;
187  case DocStyleChange::Span: /* HTML only */ break;
188  }
189 }

191 void ManDocVisitor::visit(DocVerbatim *s)
192 {
193  if (m_hide) return;
194  QCString lang = m_langExt;
195  if (!s->language().isEmpty()) // explicit language setting
196  {
197  lang = s->language();
198  }
199  SrcLangExt langExt = getLanguageFromCodeLang(lang);
200  switch (s->type())
201  {
202  case DocVerbatim::Code:
203  if (!m_firstCol) m_t << "\n";
204  m_t << ".PP\n";
205  m_t << ".nf\n";
206  getCodeParser(lang).parseCode(m_ci,s->context(),s->text(),
207  langExt,
208  s->isExample(),s->exampleFile());
209  if (!m_firstCol) m_t << "\n";
210  m_t << ".fi\n";
211  m_t << ".PP\n";
212  m_firstCol=TRUE;
213  break;
214  case DocVerbatim::Verbatim:
215  if (!m_firstCol) m_t << "\n";
216  m_t << ".PP\n";
217  m_t << ".nf\n";
218  m_t << s->text();
219  if (!m_firstCol) m_t << "\n";
220  m_t << ".fi\n";
221  m_t << ".PP\n";
222  m_firstCol=TRUE;
223  break;
224  case DocVerbatim::ManOnly:
225  m_t << s->text();
226  break;
227  case DocVerbatim::HtmlOnly:
228  case DocVerbatim::XmlOnly:
229  case DocVerbatim::LatexOnly:
230  case DocVerbatim::RtfOnly:
231  case DocVerbatim::DocbookOnly:
232  case DocVerbatim::Dot:
233  case DocVerbatim::Msc:
234  case DocVerbatim::PlantUML:
235  /* nothing */
236  break;
237  }

245 void ManDocVisitor::visit(DocInclude *inc)
246 {
247  if (m_hide) return;
248  SrcLangExt langExt = getLanguageFromFileName(inc->extension());
249  switch(inc->type())
250  {
251  case DocInclude::IncWithLines:
252  {
253  if (!m_firstCol) m_t << "\n";
254  m_t << ".PP\n";
255  m_t << ".nf\n";
256  FileInfo cfi( inc->file().str() );
257  FileDef *fd = createFileDef( cfi.dirPath(), cfi.fileName() );
258  getCodeParser(inc->extension()).parseCode(m_ci,inc->context(),
259  inc->text(),
260  langExt,
261  inc->isExample(),
262  inc->exampleFile(),
263  fd, // fileDef,
264  -1, // start line
265  -1, // end line
266  FALSE, // inline fragment
267  0, // memberDef
268  TRUE
269  );
270  delete fd;
271  if (!m_firstCol) m_t << "\n";
272  m_t << ".fi\n";
273  m_t << ".PP\n";
274  m_firstCol=TRUE;
275  }
276  break;
277  case DocInclude::Include:
278  if (!m_firstCol) m_t << "\n";
279  m_t << ".PP\n";
280  m_t << ".nf\n";
281  getCodeParser(inc->extension()).parseCode(m_ci,inc->context(),
282  inc->text(),
283  langExt,
284  inc->isExample(),
285  inc->exampleFile(),
286  0, // fileDef
287  -1, // startLine
288  -1, // endLine
289  TRUE, // inlineFragment
290  0, // memberDef
291  FALSE
292  );
293  if (!m_firstCol) m_t << "\n";
294  m_t << ".fi\n";
295  m_t << ".PP\n";
296  m_firstCol=TRUE;
297  break;
298  case DocInclude::DontInclude:
299  case DocInclude::DontIncWithLines:
300  case DocInclude::HtmlInclude:
301  case DocInclude::LatexInclude:
302  case DocInclude::RtfInclude:
303  case DocInclude::XmlInclude:
304  case DocInclude::DocbookInclude:
305  break;
306  case DocInclude::ManInclude:
307  m_t << inc->text();
308  break;
309  case DocInclude::VerbInclude:
310  if (!m_firstCol) m_t << "\n";
311  m_t << ".PP\n";
312  m_t << ".nf\n";
313  m_t << inc->text();
314  if (!m_firstCol) m_t << "\n";
315  m_t << ".fi\n";
316  m_t << ".PP\n";
317  m_firstCol=TRUE;
318  break;
319  case DocInclude::Snippet:
320  if (!m_firstCol) m_t << "\n";
321  m_t << ".PP\n";
322  m_t << ".nf\n";
323  getCodeParser(inc->extension()).parseCode(m_ci,
324  inc->context(),
325  extractBlock(inc->text(),inc->blockId()),
326  langExt,
327  inc->isExample(),
328  inc->exampleFile()
329  );
330  if (!m_firstCol) m_t << "\n";
331  m_t << ".fi\n";
332  m_t << ".PP\n";
333  m_firstCol=TRUE;
334  break;
335  case DocInclude::SnipWithLines:
336  {
337  if (!m_firstCol) m_t << "\n";
338  m_t << ".PP\n";
339  m_t << ".nf\n";
340  FileInfo cfi( inc->file().str() );
341  FileDef *fd = createFileDef( cfi.dirPath(), cfi.fileName() );
342  getCodeParser(inc->extension()).parseCode(m_ci,
343  inc->context(),
344  extractBlock(inc->text(),inc->blockId()),
345  langExt,
346  inc->isExample(),
347  inc->exampleFile(),
348  fd,
349  lineBlock(inc->text(),inc->blockId()),
350  -1, // endLine
351  FALSE, // inlineFragment
352  0, // memberDef
353  TRUE // show line number
354  );
355  delete fd;
356  if (!m_firstCol) m_t << "\n";
357  m_t << ".fi\n";
358  m_t << ".PP\n";
359  m_firstCol=TRUE;
360  }
361  break;
362  case DocInclude::SnippetDoc:
363  case DocInclude::IncludeDoc:
364  err("Internal inconsistency: found switch SnippetDoc / IncludeDoc in file: %s"
365  "Please create a bug report\n",__FILE__);
366  break;
367  }
368 }

370 void ManDocVisitor::visit(DocIncOperator *op)
371 {
372  QCString locLangExt = getFileNameExtension(op->includeFileName());
373  if (locLangExt.isEmpty()) locLangExt = m_langExt;
374  SrcLangExt langExt = getLanguageFromFileName(locLangExt);
375  //printf("DocIncOperator: type=%d first=%d, last=%d text='%s'\n",
376  // op->type(),op->isFirst(),op->isLast(),qPrint(op->text()));
377  if (op->isFirst())
378  {
379  if (!m_hide)
380  {
381  if (!m_firstCol) m_t << "\n";
382  m_t << ".PP\n";
383  m_t << ".nf\n";
384  }
385  pushHidden(m_hide);
386  m_hide = TRUE;
387  }
388  if (op->type()!=DocIncOperator::Skip)
389  {
390  m_hide = popHidden();
391  if (!m_hide)
392  {
393  FileDef *fd = 0;
394  if (!op->includeFileName().isEmpty())
395  {
396  FileInfo cfi( op->includeFileName().str() );
397  fd = createFileDef( cfi.dirPath(), cfi.fileName() );
398  }
399 
400  getCodeParser(locLangExt).parseCode(m_ci,op->context(),op->text(),langExt,
401  op->isExample(),op->exampleFile(),
402  fd, // fileDef
403  op->line(), // startLine
404  -1, // endLine
405  FALSE, // inline fragment
406  0, // memberDef
407  op->showLineNo() // show line numbers
408  );
409  if (fd) delete fd;
410  }
411  pushHidden(m_hide);
412  m_hide=TRUE;
413  }
414  if (op->isLast())
415  {
416  m_hide = popHidden();
417  if (!m_hide)
418  {
419  if (!m_firstCol) m_t << "\n";
420  m_t << ".fi\n";
421  m_t << ".PP\n";
422  m_firstCol=TRUE;
423  }
424  }
425  else
426  {
427  if (!m_hide) m_t << "\n";
428  }
429 }

679 void ManDocVisitor::visitPre(DocHtmlListItem *li)
680 {
681  if (m_hide) return;
682  QCString ws;
683  ws.fill(' ',m_indent-2);
684  if (!m_firstCol) m_t << "\n";
685  m_t << ".IP \"" << ws;
686  if (((DocHtmlList *)li->parent())->type()==DocHtmlList::Ordered)
687  {
688  for (const auto &opt : li->attribs())
689  {
690  if (opt.name=="value")
691  {
692  bool ok;
693  int val = opt.value.toInt(&ok);
694  if (ok) man_listItemInfo[m_indent].number = val;
695  }
696  }
697  switch (man_listItemInfo[m_indent].type)
698  {
699  case '1':
700  m_t << man_listItemInfo[m_indent].number;
701  break;
702  case 'a':
703  m_t << integerToAlpha(man_listItemInfo[m_indent].number,false);
704  break;
705  case 'A':
706  m_t << integerToAlpha(man_listItemInfo[m_indent].number);
707  break;
708  case 'i':
709  m_t << integerToRoman(man_listItemInfo[m_indent].number,false);
710  break;
711  case 'I':
712  m_t << integerToRoman(man_listItemInfo[m_indent].number);
713  break;
714  default:
715  m_t << man_listItemInfo[m_indent].number;
716  break;
717  }
718  m_t << ".\" " << m_indent+2;
719  man_listItemInfo[m_indent].number++;
720  }
721  else // bullet list
722  {
723  m_t << "\\(bu\" " << m_indent;
724  }
725  m_t << "\n";
726  m_firstCol=TRUE;
727 }
