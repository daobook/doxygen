542 void PerlModDocVisitor::visit(DocSymbol *sy)
543 {
544  const DocSymbol::PerlSymb *res = HtmlEntityMapper::instance()->perl(sy->symbol());
545  const char *accent=0;
546  if (res-> symb)
547  {
548  switch (res->type)
549  {
550  case DocSymbol::Perl_string:
551  enterText();
552  m_output.add(res->symb);
553  break;
554  case DocSymbol::Perl_char:
555  enterText();
556  m_output.add(res->symb[0]);
557  break;
558  case DocSymbol::Perl_symbol:
559  leaveText();
560  openItem("symbol");
561  m_output.addFieldQuotedString("symbol", res->symb);
562  closeItem();
563  break;
564  default:
565  switch(res->type)
566  {
567  case DocSymbol::Perl_umlaut:
568  accent = "umlaut";
569  break;
570  case DocSymbol::Perl_acute:
571  accent = "acute";
572  break;
573  case DocSymbol::Perl_grave:
574  accent = "grave";
575  break;
576  case DocSymbol::Perl_circ:
577  accent = "circ";
578  break;
579  case DocSymbol::Perl_slash:
580  accent = "slash";
581  break;
582  case DocSymbol::Perl_tilde:
583  accent = "tilde";
584  break;
585  case DocSymbol::Perl_cedilla:
586  accent = "cedilla";
587  break;
588  case DocSymbol::Perl_ring:
589  accent = "ring";
590  break;
591  default:
592  break;
593  }
594  leaveText();
595  if (accent)
596  {
597  openItem("accent");
598  m_output
599  .addFieldQuotedString("accent", accent)
600  .addFieldQuotedChar("letter", res->symb[0]);
601  closeItem();
602  }
603  break;
604  }
605  }
606  else
607  {
608  err("perl: non supported HTML-entity found: %s\n",HtmlEntityMapper::instance()->html(sy->symbol(),TRUE));
609  }
